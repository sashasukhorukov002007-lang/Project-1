#include <iostream>
#include <set>
#include <map>
#include <sstream>
#include <string>
using namespace std;

void taskSet() {
// Создаём множество целых чисел с начальными значениями
// set автоматически сортирует элементы и хранит только уникальные значения
    set<int> s = { 5, 2, 8, 2, 3, 5, 1 };
// Фактически в множестве сохранятся: 1, 2, 3, 5, 8 (повторения 2 и 5 будут проигнорированы)

    cout << "Исходное множество: ";
// Проходим по всем элементам множества (они уже отсортированы)
    for (int x : s)
    cout << x << " ";
    cout <<endl;

    int value;
   cout << "Введите число: ";
   cin >> value;
// Проверяем, есть ли введённое число в множестве
// find() возвращает итератор на найденный элемент или s.end(), если элемент не найден

    if (s.find(value) != s.end()) {
        s.erase(value);
        cout << "Число " << value << " удалено.\n";
    }
    else {
        s.insert(value);
        cout << "Число " << value << " добавлено.\n";
    }

    cout << "Обновлённое множество: ";
    for (int x : s)
    cout << x << " ";
    cout << endl;

    cin.ignore(); // Очищаем буфер ввода от символа новой строки после cin >>
    string line;
    cout << "\nВведите строку: ";
    getline(cin, line); // Считываем всю строку целиком (может содержать пробелы)

// Создаём поток для разбора строки на отдельные слова
    stringstream ss(line);
    set<string> uniqueWords; // Множество для хранения уникальных слов
    string word;

// Извлекаем слова из потока (разделителем служит пробел)
    while (ss >> word)
        uniqueWords.insert(word); // Добавляем слово в множество (повторы автоматически игнорируются)

    cout << "Уникальных слов: " << uniqueWords.size() << std::endl;
// size() возвращает количество уникальных элементов в множестве
}
